<Meta title="Colors" />

export const StyledSpan = styled('span', {
  background: '$slateGray200',
  color: '$secondaryA800',
  padding: '5px 8px',
  borderRadius: '3px'
})

export const colorValues = re => {
  return Object.keys(theme.colors)
    .filter(color => re.test(color))
    .map(key => theme.colors[key].value)
}

# Colors

Colors are typically accessed with [Stitches token syntax](https://stitches.dev/docs/tokens). Colors documented below also have an alpha channel which is accessed by prefixing the swatch with a capital A.

## Example

```ts
import { styled } from '@junat/design'

const StyledSpan = styled('span', {
  background: '$slateGray200',
  color: '$secondaryA800',
  padding: '5px 8px',
  borderRadius: '3px'
})
```

Will render as: <StyledSpan>Text</StyledSpan>
